pipeline { agent any options { disableConcurrentBuilds() } environment { PROJECT_NAME = "simpleweb_v2" BASE_DIR = "/opt/projects" PROJECT_DIR = "${BASE_DIR}/${PROJECT_NAME}" NGINX_DIR = "/var/www/html" REPO_URL = "https://github.com/vijayqadevkraft/webapplication-repo.git" } stages { stage('Checkout Code') { steps { echo "📥 Checking out code..." dir("${PROJECT_DIR}") { git branch: 'main', url: "${REPO_URL}" } } } stage('Check simpleweb Changes') { when { changeset "simpleweb/**" } steps { echo "✅ simpleweb changed → Continue pipeline" } } stage('Setup Permissions') { steps { sh ''' sudo mkdir -p $BASE_DIR sudo chown -R jenkins:jenkins $BASE_DIR ''' } } stage('Validate File') { steps { sh ''' ls $PROJECT_DIR/simpleweb if [ ! -f "$PROJECT_DIR/simpleweb/index.html" ]; then echo "❌ index.html not found!" exit 1 fi ''' } } stage('Deploy to Nginx') { steps { sh ''' sudo rm -rf $NGINX_DIR/* sudo cp -r $PROJECT_DIR/simpleweb/* $NGINX_DIR/ ''' } } stage('Reload Nginx') { steps { sh ''' sudo systemctl reload nginx ''' } } stage('Verify Deployment') { steps { sh ''' curl -I http://localhost ''' } } } post { success { echo "🎉 Deployment Successful!" } failure { echo "❌ Deployment Failed!" } } }
